# Phase 5: Visual Captioning & Table of Contents (ToC)

## Overview
In this phase, we ingest the JSON array of event triggers generated by the Phase 4 Sentry. For each timestamp identified as an "Interest Zone", we extract the corresponding visual frame from the source video. We then pass this frame through Qwen2.5-VL-3B (via MLX) to generate a dense semantic caption of the user's action. The output is a highly structured Table of Contents (ToC) that **merges** acoustic trigger captions with the Phase 2 event annotations to ensure full temporal coverage.

## Optimization for Apple Silicon
To maintain speed without sacrificing capability, we use the `mlx-vlm` library to run Moondream2. The models should be cached on the 2TB SSD to avoid clogging the internal M4 Mac storage.

> [!NOTE]
> **Caption Quality Consideration:** Moondream2 (~1.8B params) is optimized for speed but may produce generic captions for blurry/occluded egocentric frames. If caption quality proves insufficient during validation, consider upgrading to a quantized Qwen2.5-VL-7B (via `mlx-vlm`) for a subset of critical triggers, or using multi-frame context.

## Trigger-Aware Prompting
Different trigger types imply different temporal contexts. We adapt the VLM prompt accordingly:

| Trigger Type | Prompt Variant |
|---|---|
| `spike` (action start) | "What action is the user starting to perform with their hands? Be specific about tools and objects." |
| `drop` (action end) | "What action did the user just finish performing? Describe the result of the completed action." |
| `forced_silence` | "What is the user currently doing? Describe the ongoing activity and any objects being manipulated." |

## Pseudocode Implementation

```python
import cv2
import json
import os
from mlx_vlm import load, generate

# System Configuration
SSD_BASE = "/Volumes/Extreme SSD/goal_step_data"
MODEL_CACHE = os.path.join(SSD_BASE, "models/moondream2")

# Trigger-aware prompts for better caption relevance
PROMPTS = {
    "spike": "What action is the user starting to perform with their hands? Be specific about tools and objects.",
    "drop": "What action did the user just finish performing? Describe the result of the completed action.",
    "forced_silence": "What is the user currently doing? Describe the ongoing activity and any objects being manipulated."
}

class CaptioningEngine:
    def __init__(self):
        # Load Moondream2 using MLX optimizations
        print("Loading Moondream2 into unified memory...")
        os.environ['HF_HOME'] = MODEL_CACHE # Force HuggingFace to use SSD
        self.model, self.processor = load("vikhyatk/moondream2")

    def extract_frame(self, video_path: str, timestamp_s: float, video_id: str) -> str:
        """ 
        Extracts a specific frame and saves it to the SSD with a unique filename.
        Returns the path to the saved image.
        """
        cap = cv2.VideoCapture(video_path)
        
        # Seek to the requested timestamp (in milliseconds)
        cap.set(cv2.CAP_PROP_POS_MSEC, timestamp_s * 1000)
        ret, frame = cap.read()
        cap.release()
        
        if not ret:
            raise ValueError(f"Could not read frame at timestamp {timestamp_s}")
        
        # Use unique filename per (video, timestamp) to avoid race conditions
        safe_ts = str(timestamp_s).replace('.', '_')
        img_path = os.path.join(SSD_BASE, f"cache/phase5/frames/{video_id}_{safe_ts}.jpg")
        os.makedirs(os.path.dirname(img_path), exist_ok=True)
        cv2.imwrite(img_path, frame)
        return img_path
    
    def extract_multi_frame_context(self, video_path: str, timestamp_s: float, video_id: str, window_sec: float = 0.5) -> list:
        """
        Extract 3 frames around the trigger point for richer context.
        Returns list of image paths: [before, at, after].
        """
        timestamps = [
            max(0, timestamp_s - window_sec),
            timestamp_s,
            timestamp_s + window_sec
        ]
        paths = []
        for ts in timestamps:
            try:
                path = self.extract_frame(video_path, ts, video_id)
                paths.append(path)
            except ValueError:
                pass  # End of video, skip
        return paths

    def process_video_toc(self, video_id: str):
        video_path = os.path.join(SSD_BASE, f"videos/{video_id}.mp4")
        trigger_path = os.path.join(SSD_BASE, f"cache/phase4/{video_id}_triggers.json")
        
        if not os.path.exists(trigger_path):
            raise FileNotFoundError(f"Triggers for {video_id} not found.")
            
        with open(trigger_path, 'r') as f:
            triggers = json.load(f)
            
        toc_entries = []
        os.makedirs(os.path.join(SSD_BASE, "cache/phase5/frames"), exist_ok=True)
        
        print(f"Processing {len(triggers)} triggers for {video_id}...")
        
        for t in triggers:
            timestamp = t['timestamp']
            trigger_type = t['trigger_type']
            
            # 1. Grab visual context (primary frame)
            try:
                img_path = self.extract_frame(video_path, timestamp, video_id)
            except ValueError as e:
                print(f"[SKIP] Frame extraction failed at {timestamp}s: {e}")
                continue
            
            # 2. Select trigger-aware prompt
            prompt = PROMPTS.get(trigger_type, PROMPTS["forced_silence"])
            
            # 3. Run Moondream inference
            try:
                caption_out = generate(
                    self.model, 
                    self.processor, 
                    img_path, 
                    prompt, 
                    max_tokens=64,
                    temperature=0.2
                )
                caption_text = caption_out.strip()
            except Exception as e:
                print(f"[ERROR] Moondream inference failed at {timestamp}s: {e}")
                caption_text = "[CAPTION_FAILED]"
            
            # 4. Quality filter: skip obviously generic/failed captions
            if len(caption_text) < 5 or caption_text == "[CAPTION_FAILED]":
                print(f"[WARN] Low quality caption at {timestamp}s, retrying with default prompt...")
                try:
                    caption_out = generate(
                        self.model, self.processor, img_path,
                        "Describe exactly what is happening in this image.",
                        max_tokens=64, temperature=0.3
                    )
                    caption_text = caption_out.strip()
                except Exception:
                    caption_text = "Unable to caption frame"
            
            # 5. Construct ToC Entry
            entry = {
                "timestamp": timestamp,
                "trigger_type": trigger_type,
                "caption": caption_text
            }
            toc_entries.append(entry)
            print(f"[{timestamp}s | {trigger_type}]: {entry['caption']}")
            
        # 6. Merge with Phase 2 Event Annotations
        event_log_path = os.path.join(SSD_BASE, f"cache/phase2/{video_id}_events.json")
        if os.path.exists(event_log_path):
            with open(event_log_path, 'r') as f:
                event_log = json.load(f)
            # Add event annotations as ToC entries with source="event_sample"
            for event in event_log:
                toc_entries.append({
                    "timestamp": event["start_time"],
                    "end_time": event["end_time"],
                    "trigger_type": "event_sample",
                    "caption": event["caption"]
                })
            print(f"  Merged {len(event_log)} event annotations from Phase 2")
        
        # Sort by timestamp for chronological ordering
        toc_entries.sort(key=lambda x: x["timestamp"])
        
        # 7. Save Final ToC
        toc_out_path = os.path.join(SSD_BASE, f"cache/phase5/{video_id}_toc.json")
        with open(toc_out_path, 'w') as f:
            json.dump(toc_entries, f, indent=4)
        
        print(f"ToC successfully constructed for {video_id}. ({len(toc_entries)} entries)")

if __name__ == '__main__':
    engine = CaptioningEngine()
    # Example: run on an EPIC-KITCHENS video
    engine.process_video_toc("P01_101")
```

## Verification Strategy
- **Spot-Check Frame I/O:** Frames are now saved with unique filenames per (video, timestamp). Manually inspect 10 saved JPEGs to confirm they match the expected video moments.
- **Prompt Differentiation Test:** Process a video with known spike/drop/silence triggers. Verify that "spike" captions describe *starting* actions while "drop" captions describe *completed* actions.
- **Hallucination Rate:** Read through the generated ToC JSON. Check if Moondream2 is consistently returning generic phrases (e.g. "a person's hand") vs highly descriptive steps ("user is turning a wrench"). If generic, evaluate upgrading to Qwen2.5-VL-7B for critical frames.
- **Caption Coverage:** Assert that `len(toc_entries) / len(triggers) > 0.95` — i.e., fewer than 5% of triggers fail to produce a usable caption.
